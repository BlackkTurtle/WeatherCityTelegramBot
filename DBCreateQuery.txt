CREATE TABLE Users (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    UserId NVARCHAR(50) NOT NULL UNIQUE,
    StartDate DATETIME NOT NULL
);

CREATE TABLE WeatherHistories (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    City NVARCHAR(100) NOT NULL,
    CreationDate DATETIME NOT NULL,
    UserId INT NOT NULL,
    CONSTRAINT FK_WeatherHistory_Users FOREIGN KEY (UserId) REFERENCES Users(Id) ON DELETE CASCADE
);
